function Button({ children, className = "", variant = "primary" }) {
  const base =
    "inline-flex items-center justify-center rounded-full px-8 py-4 text-base font-bold transition focus:outline-none focus:ring-2 focus:ring-cyan-300 focus:ring-offset-2 focus:ring-offset-slate-950";

      : "bg-cyan-400 text-slate-950 hover:bg-cyan-300";


  return <span className={`${base} ${styles} ${className}`}>{children}</span>;
}


function Card({ children, className = "" }) {
